 'Y' -> 2019 (Year full)
 'y' -> 19 (Year short)
 'M' -> July (Month full)
 'm' -> Jul (Month short)
 'D' -> 01 (date padded)
 'd' -> 1 (date)
 'H' -> 05 (Hours padded)
 'h' -> 5 (Hours)
 'I' -> 08 (Minutes padded)
 'i' -> 8 (Minutes)
 'S' -> 04 (Seconds padded)
 's' -> 4 (Seconds)
